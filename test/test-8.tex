  [latin][default]
  [onum=yes,pnum=yes,protrusion=pure,expansion=quality]

  [myfonts][serif][nsimsun]
  [features=hanzi,force=yes,preset=range:chinese]
\definefontfamily[myfonts][serif][texgyrepagella][features=latin]
\setupbodyfont[myfonts]
